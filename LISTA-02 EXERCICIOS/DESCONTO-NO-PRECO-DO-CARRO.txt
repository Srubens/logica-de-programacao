Var

NOMEDOCARRO:CARACTER
VALORDOCARRO, DESCONTO:REAL
ANODOCARRO:INTEIRO


Inicio


ESCREVA("INFORME O NOME DO CARRO QUE DESEJA COMPRAR: ")
LEIA(NOMEDOCARRO)
ESCREVA("INFORME O ANO DO CARRO: ")
LEIA(ANODOCARRO)
ESCREVA("INFORME O VALOR DO CARRO: R$")
LEIA(VALORDOCARRO)

SE ANODOCARRO <= 2000 ENTAO
   DESCONTO <- VALORDOCARRO - (VALORDOCARRO * 12/100)
   ESCREVA("O CARRO ", NOMEDOCARRO, " SAI POR R$ ", DESCONTO)
SENAO
   SE ANODOCARRO >= 2000 ENTAO
      DESCONTO <- VALORDOCARRO - (VALORDOCARRO * 7/100)
      ESCREVA("O CARRO ", NOMEDOCARRO, " SAI POR R$ ", DESCONTO)
   FIMSE
FIMSE


Fimalgoritmo
